/* Create the tables: */

    rank INT NOT NULL,
    score DOUBLE NOT NULL,
    standError DOUBLE,
    economy DOUBLE,
    family DOUBLE,
    health DOUBLE,
    freedom DOUBLE,
    trust DOUBLE,
    generosity DOUBLE,
    dystopia DOUBLE
);

    generosity DOUBLE,

);
